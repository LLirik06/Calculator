Вот код js(поповоду плюса перед prompt, ты не обьяснил как делать сумму через a+b, поэтому я загуглил как сделать сумму и все)

let a = +prompt("Введите первое число")

let b = +prompt("Введите второе число")

summa = `Сумма этих чисел, ${a+b}`

minus = `Разность этих чисел, ${a-b}`

umnoj = `Произведение этих чисел, ${a*b}`

delenie = `Частное этих чисел ${a/b}`

alert(summa)

alert(minus)

alert(umnoj)

alert(delenie)
